## Fast URL Checker Script (Python 3.7+)

This script uses `aiohttp` for concurrent requests and `argparse` for flexible command-line options.

### 1. Prerequisites

You need the `aiohttp` library installed:

### 2. The Script (`scanner.py`)

Save the following code as `scanner.py`:

```python
import asyncio
import aiohttp
import argparse
import sys

# ANSI color codes for better readability
GREEN = "\033[92m"
RED = "\033[91m"
YELLOW = "\033[93m"
RESET = "\033[0m"

async def check_url(session, url, keyword, timeout):
    """
    Performs an asynchronous GET request and checks for the keyword.
    """
    if not url.startswith(('http://', 'https://')):
        url = 'http://' + url # Default to HTTP if no scheme is specified

    keyword_lower = keyword.lower()
    
    try:
        # Set specific timeout for the request
        conn_timeout = aiohttp.ClientTimeout(total=timeout)
        
        async with session.get(url, timeout=conn_timeout, allow_redirects=True) as response:
            status_code = response.status
            
            # Only read content if the status is generally successful
            if 200 <= status_code < 400:
                content = await response.text()
                
                if keyword_lower in content.lower():
                    print(f"{GREEN}[FOUND]{RESET} | Status: {status_code} | URL: {url} | Keyword '{keyword}'")
                    return url, True
                else:
                    print(f"[MISS]  | Status: {status_code} | URL: {url}")
                    return url, False
            else:
                # Handle non-200 status codes (e.g., 404, 500)
                print(f"[FAIL]  | Status: {status_code} | URL: {url}")
                return url, False

    except aiohttp.ClientConnectorError:
        # Handles DNS failures, connection refused, etc.
        print(f"{YELLOW}[ERROR]{RESET} | Connection Error | URL: {url}")
    except asyncio.TimeoutError:
        # Handles connection timeouts
        print(f"{YELLOW}[ERROR]{RESET} | Timeout ({timeout}s)    | URL: {url}")
    except Exception as e:
        # Catch any other general errors (e.g., invalid headers, parsing errors)
        print(f"{RED}[ERROR]{RESET} | Unexpected Error: {type(e).__name__} | URL: {url}")
        
    return url, None


async def run_checks(urls, keyword, timeout, concurrency_limit):
    """
    Manages the creation and execution of concurrent tasks.
    """
    # Use a Semaphore to limit the number of simultaneous connections
    semaphore = asyncio.Semaphore(concurrency_limit)

    async def check_with_semaphore(url):
        async with semaphore:
            return await check_url(session, url, keyword, timeout)

    # aiohttp session must be created inside the async function
    # The session is responsible for connection pooling and header management
    async with aiohttp.ClientSession() as session:
        tasks = [check_with_semaphore(url) for url in urls]
        await asyncio.gather(*tasks)


def read_urls(filename):
    """Reads URLs from the specified file."""
    try:
        with open(filename, 'r') as f:
            # Cleanly strip whitespace and ignore empty lines
            urls = [line.strip() for line in f if line.strip()]
        return urls
    except FileNotFoundError:
        print(f"{RED}Error: File '{filename}' not found.{RESET}", file=sys.stderr)
        sys.exit(1)


def main():
    parser = argparse.ArgumentParser(
        description="Fast Asynchronous URL Keyword Scanner.",
        formatter_class=argparse.RawTextHelpFormatter
    )
    parser.add_argument(
        '-f', '--file',
        type=str,
        default='urls.txt',
        help='Input file containing URLs (one per line). Default: urls.txt'
    )
    parser.add_argument(
        '-k', '--keyword',
        type=str,
        default='Admin',
        help='The keyword to search for in the response content. Default: Admin'
    )
    parser.add_argument(
        '-t', '--timeout',
        type=int,
        default=5,
        help='Connection timeout in seconds. Default: 5'
    )
    parser.add_argument(
        '-c', '--concurrency',
        type=int,
        default=100,
        help='Maximum number of simultaneous connections. Default: 100'
    )
    
    args = parser.parse_args()

    urls = read_urls(args.file)
    if not urls:
        print("No URLs found in the file. Exiting.")
        return

    print(f"--- Starting Scan ---")
    print(f"File: {args.file} | Keyword: '{args.keyword}' | URLs to check: {len(urls)}")
    print(f"Concurrency: {args.concurrency} | Timeout: {args.timeout}s")
    print("-" * 30)

    # Start the asyncio event loop
    try:
        asyncio.run(run_checks(urls, args.keyword, args.timeout, args.concurrency))
    except KeyboardInterrupt:
        print(f"\n{YELLOW}Scan interrupted by user.{RESET}")


if __name__ == "__main__":
    main()
```
